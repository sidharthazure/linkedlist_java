# Project built for internal evaluation
$ git branch                             (This will list all the brach)
$ git branch new_branch                  (This will create the new brach as name of new_branch which will be on local)
$ git push origin new_branch             (This will put the new_branch remotely)
$ git checkout new_branch
$ git checkout master
$ git branch -d new_branch               (This command will delete the new_branch, it will be deleted locally)
$ git push origin --delete new_branch    (This will delete remotely)
